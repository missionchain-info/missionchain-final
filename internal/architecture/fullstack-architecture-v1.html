// ── Generator & Datasource ──────────────────────────────────────

// ── ENUMS ───────────────────────────────────────────────────────
enum UserRole {
  USER
  CREATOR
  ADMIN
  SUPER_ADMIN
}

enum CreatorTier {
  BUILDER    // Entry level
  MAKER      // Intermediate
  LUMINARY   // Top tier
}

enum SeedPackage {
  EARLY_BIRD       // $1,000 — 400,000 MIC + 20 MFP-NFT
  FOUNDING_I       // $2,500 — 1,000,000 MIC + 60 MFP-NFT
  FOUNDING_II      // $5,000 — 2,000,000 MIC + 150 MFP-NFT
  FOUNDING_III     // $10,000 — 4,000,000 MIC + 350 MFP-NFT
}

enum SeedStatus {
  PENDING          // Awaiting on-chain verification
  VERIFYING        // Being checked
  VERIFIED         // TXID confirmed on BSCScan
  FULFILLED        // MIC + NFT delivered
  REJECTED         // Invalid TXID or wrong amount
}

enum MICEStatus {
  ACTIVE
  EXPIRED
  RECYCLED
}

enum ContentType {
  PODCAST
  ARTICLE
  VIDEO_SHORT
  DEVOTIONAL    // SOPHIA WORD
}

enum ContentLang {
  EN
  ES
  PT
  KO
  VI
}

enum ChallengeCategory {
  WORSHIP_MUSIC
  BIBLE_ART
  TESTIMONY_FILM
  AI_CONTENT
  DESIGN
  WRITING
}

enum ChallengeStatus {
  DRAFT
  OPEN
  VOTING
  JUDGING
  COMPLETED
  CANCELLED
}

enum JobStatus {
  OPEN
  IN_PROGRESS
  COMPLETED
  CANCELLED
}

enum NFTCredentialType {
  BUILDER
  MAKER
  LUMINARY
  MFP            // Mission Founding Partner (governance)
  COMMUNITY      // Community NFT
}

enum NotificationType {
  SEED_VERIFIED
  SEED_FULFILLED
  CHALLENGE_RESULT
  JOB_MATCHED
  CREDENTIAL_EARNED
  SYSTEM
}

// ── USERS ───────────────────────────────────────────────────────
model User {
  id            String     @id @default(cuid())
  email         String     @unique
  emailVerified Boolean    @default(false)
  username      String?    @unique
  displayName   String?
  bio           String?
  avatarUrl     String?
  coverUrl      String?
  role          UserRole   @default(USER)
  tier          CreatorTier?
  country       String?    // ISO 3166-1 alpha-2
  language      String?    @default("en")
  
  // Gamification
  points        Int        @default(0)
  jobsCompleted Int        @default(0)
  rating        Decimal?   @db.Decimal(3,2)
  ratingCount   Int        @default(0)
  
  // Social links
  websiteUrl    String?
  twitterHandle String?
  telegramHandle String?
  
  // Auth
  passwordHash  String?    // null if wallet-only login
  lastLoginAt   DateTime?
  
  createdAt     DateTime   @default(now())
  updatedAt     DateTime   @updatedAt
  
  // Relations
  wallets           Wallet[]
  seedParticipations SeedParticipant[]
  miceHoldings      MICELicense[]
  nftCredentials    NFTCredential[]
  creatorProfile    CreatorProfile?
  contentItems      Content[]
  challengeEntries  ChallengeEntry[]
  jobPostings       Job[]         @relation("JobPoster")
  jobApplications   JobApplication[]
  notifications     Notification[]
  sessions          UserSession[]
  
  @@map("users")
}

// ── USER SESSIONS ───────────────────────────────────────────────
model UserSession {
  id          String   @id @default(cuid())
  userId      String
  user        User     @relation(fields: [userId], references: [id], onDelete: Cascade)
  token       String   @unique
  refreshToken String? @unique
  userAgent   String?
  ipAddress   String?
  expiresAt   DateTime
  createdAt   DateTime @default(now())
  
  @@map("user_sessions")
}

// ── WALLETS ─────────────────────────────────────────────────────
model Wallet {
  id          String   @id @default(cuid())
  userId      String
  user        User     @relation(fields: [userId], references: [id], onDelete: Cascade)
  address     String   @unique   // BEP20 address (checksummed)
  isPrimary   Boolean  @default(false)
  label       String?
  verifiedAt  DateTime?          // Signature verification timestamp
  createdAt   DateTime @default(now())
  
  seedParticipations SeedParticipant[] @relation("RecipientWallet")
  
  @@map("wallets")
}

// ── SEED ROUND ──────────────────────────────────────────────────
model SeedParticipant {
  id              String       @id @default(cuid())
  userId          String?      // null if submitted before account creation
  user            User?        @relation(fields: [userId], references: [id])
  
  // Contact
  email           String
  
  // Package
  package         SeedPackage
  amountUSDT      Decimal      @db.Decimal(12,2)
  micAllocation   BigInt       // in MIC wei (18 decimals)
  mfpNftCount     Int
  
  // Blockchain
  txid            String       @unique
  senderAddress   String?      // detected from BSCScan
  recipientWalletId String?
  recipientWallet   Wallet?    @relation("RecipientWallet", fields: [recipientWalletId], references: [id])
  recipientAddress  String     // BEP20 address for delivery
  
  // Status
  status          SeedStatus   @default(PENDING)
  verifiedAt      DateTime?
  fulfilledAt     DateTime?
  rejectionReason String?
  
  // Admin notes
  adminNotes      String?
  verifiedBy      String?      // Admin user ID
  
  createdAt       DateTime     @default(now())
  updatedAt       DateTime     @updatedAt
  
  @@index([email])
  @@index([status])
  @@index([txid])
  @@map("seed_participants")
}

// ── MICE LICENSES ───────────────────────────────────────────────
model MICELicense {
  id              String     @id @default(cuid())
  userId          String
  user            User       @relation(fields: [userId], references: [id])
  
  licenseNumber   Int        @unique  // Sequential 1-100000
  walletAddress   String
  
  // Pricing at activation
  priceUSDT       Decimal    @db.Decimal(8,2)
  priceMIC        BigInt     // 50% MIC component
  liquidityBand   String     // "< $200K", "$200K-$1M", etc.
  
  status          MICEStatus @default(ACTIVE)
  activatedAt     DateTime   @default(now())
  expiresAt       DateTime   // activatedAt + 360 days
  recycledAt      DateTime?
  
  // Mining stats (updated by cron)
  totalMined      BigInt     @default(0)
  lastMineAt      DateTime?
  
  purchaseTxid    String?
  
  createdAt       DateTime   @default(now())
  updatedAt       DateTime   @updatedAt
  
  @@index([userId])
  @@index([status])
  @@map("mice_licenses")
}

// ── NFT CREDENTIALS ─────────────────────────────────────────────
model NFTCredential {
  id              String            @id @default(cuid())
  userId          String
  user            User              @relation(fields: [userId], references: [id])
  
  type            NFTCredentialType
  tokenId         BigInt?           // On-chain token ID (null until minted)

  metadataUri     String?           // IPFS metadata JSON URI
  
  // Earn condition
  earnedReason    String?
  issuedAt        DateTime?
  expiresAt       DateTime?         // annual re-verification
  
  mintedAt        DateTime?
  mintTxid        String?
  
  createdAt       DateTime          @default(now())
  
  @@index([userId])
  @@index([type])
  @@map("nft_credentials")
}

// ── JOBS (MISSION WORK) ─────────────────────────────────────────
model Job {
  id              String        @id @default(cuid())
  posterId        String
  poster          User          @relation("JobPoster", fields: [posterId], references: [id])
  
  title           String
  description     String
  category        String        // "Music Production", "Design", etc.
  
  // Compensation
  budgetMIC       BigInt?
  budgetType      String        // "fixed", "hourly", "monthly"
  
  // Requirements
  requiredTier    CreatorTier?  // Minimum tier to apply
  requiredSkills  String[]
  locationType    String        // "remote", "hybrid", "onsite"
  location        String?
  
  status          JobStatus     @default(OPEN)
  applicationCount Int          @default(0)
  
  expiresAt       DateTime?
  createdAt       DateTime      @default(now())
  updatedAt       DateTime      @updatedAt
  
  applications    JobApplication[]
  
  @@index([status, category])
  @@map("jobs")
}

model JobApplication {
  id          String   @id @default(cuid())
  jobId       String
  job         Job      @relation(fields: [jobId], references: [id])
  applicantId String
  applicant   User     @relation(fields: [applicantId], references: [id])
  
  coverLetter String?
  proposedMIC BigInt?
  
  status      String   @default("pending")  // pending, accepted, rejected
  reviewedAt  DateTime?
  
  createdAt   DateTime @default(now())
  
  @@unique([jobId, applicantId])
  @@map("job_applications")
}

// ── NOTIFICATIONS ───────────────────────────────────────────────

  readAt    DateTime?

  @@index([userId, isRead])
  @@map("notifications")
}

import "@openzeppelin/contracts/access/Ownable.sol";

/**
 * @title SeedSale
 * @notice Handles USDT payments for MIC seed round
 * @dev Records purchases on-chain. Actual MIC/NFT delivery is off-chain via admin.
 *      This contract serves as transparent proof of payment.
 */
contract SeedSale is Ownable, ReentrancyGuard, Pausable {
    IERC20 public immutable usdt;
    address public treasury;

    uint256 public constant SEED_PRICE = 25;           // $0.0025 = 25 * 10^-4
    uint256 public constant PRICE_DENOMINATOR = 10000;

    // Package definitions
    struct Package {
        uint256 usdtAmount;    // in USDT wei (6 decimals on BSC)
        uint256 micAmount;     // in MIC wei (18 decimals)
        uint256 mfpNftCount;
        bool active;
    }

    mapping(uint8 => Package) public packages;

    struct Purchase {
        address buyer;
        uint8 packageId;
        uint256 usdtPaid;
        uint256 micAllocated;
        uint256 mfpNfts;
        uint256 timestamp;
        bool fulfilled;
    }

    Purchase[] public purchases;
    mapping(address => uint256[]) public buyerPurchases;

    uint256 public totalRaised;     // USDT raised
    uint256 public totalMICAllocated;

    event PurchaseCreated(
        uint256 indexed purchaseId,
        address indexed buyer,
        uint8 packageId,
        uint256 usdtPaid,
        uint256 micAllocated,
        uint256 mfpNfts
    );
    event PurchaseFulfilled(uint256 indexed purchaseId);
    event TreasuryUpdated(address indexed newTreasury);

    constructor(address _usdt, address _treasury) Ownable(msg.sender) {
        usdt = IERC20(_usdt);
        treasury = _treasury;
        _initPackages();
    }

    function _initPackages() internal {
        // EARLY_BIRD: $1,000
        packages[0] = Package(1000 * 10**18, 400_000_000 * 10**18, 20, true);
        // FOUNDING_I: $2,500  
        packages[1] = Package(2500 * 10**18, 1_000_000_000 * 10**18, 60, true);
        // FOUNDING_II: $5,000
        packages[2] = Package(5000 * 10**18, 2_000_000_000 * 10**18, 150, true);
        // FOUNDING_III: $10,000
        packages[3] = Package(10000 * 10**18, 4_000_000_000 * 10**18, 350, true);
    }

    function purchase(uint8 packageId) external nonReentrant whenNotPaused {
        Package memory pkg = packages[packageId];
        require(pkg.active, "SeedSale: Package not active");

        bool sent = usdt.transferFrom(msg.sender, treasury, pkg.usdtAmount);
        require(sent, "SeedSale: USDT transfer failed");

        uint256 purchaseId = purchases.length;
        purchases.push(Purchase({
            buyer: msg.sender,
            packageId: packageId,
            usdtPaid: pkg.usdtAmount,
            micAllocated: pkg.micAmount,
            mfpNfts: pkg.mfpNftCount,
            timestamp: block.timestamp,
            fulfilled: false
        }));

        buyerPurchases[msg.sender].push(purchaseId);
        totalRaised += pkg.usdtAmount;
        totalMICAllocated += pkg.micAmount;

        emit PurchaseCreated(purchaseId, msg.sender, packageId,
            pkg.usdtAmount, pkg.micAmount, pkg.mfpNftCount);
    }

    function markFulfilled(uint256 purchaseId) external onlyOwner {
        require(purchaseId < purchases.length, "SeedSale: Invalid id");
        purchases[purchaseId].fulfilled = true;
        emit PurchaseFulfilled(purchaseId);
    }

    function getBuyerPurchases(address buyer) external view returns (uint256[] memory) {
        return buyerPurchases[buyer];
    }

    function getTotalPurchases() external view returns (uint256) {
        return purchases.length;
    }

    function updateTreasury(address _treasury) external onlyOwner {
        treasury = _treasury;
        emit TreasuryUpdated(_treasury);
    }

    function pause() external onlyOwner { _pause(); }
    function unpause() external onlyOwner { _unpause(); }
}

pragma solidity ^0.8.24;

import "@openzeppelin/contracts/token/ERC721/ERC721.sol";

import "@openzeppelin/contracts/utils/Pausable.sol";

/**
 * @title MFPNft
 * @notice Mission Founding Partner NFT — governance credential
 * @dev 25,000 hard cap. Non-transferable (soulbound) option available per token.
 */
contract MFPNft is ERC721, ERC721URIStorage, ERC721Enumerable, AccessControl, Pausable {
    bytes32 public constant MINTER_ROLE = keccak256("MINTER_ROLE");

    uint256 public constant MAX_SUPPLY = 25_000;
    uint256 private _nextTokenId;

    // Soulbound: tokens marked as non-transferable
    mapping(uint256 => bool) public soulbound;

    event NFTMinted(address indexed to, uint256 tokenId, string uri);
    event SoulboundSet(uint256 tokenId, bool isSoulbound);

    constructor() ERC721("Mission Founding Partner NFT", "MFP") {
        _grantRole(DEFAULT_ADMIN_ROLE, msg.sender);
        _grantRole(MINTER_ROLE, msg.sender);
    }

    function mint(address to, string calldata uri, bool _soulbound) 
        external onlyRole(MINTER_ROLE) whenNotPaused 
    {
        require(_nextTokenId < MAX_SUPPLY, "MFPNft: Max supply reached");
        uint256 tokenId = _nextTokenId++;
        _safeMint(to, tokenId);
        _setTokenURI(tokenId, uri);
        soulbound[tokenId] = _soulbound;
        emit NFTMinted(to, tokenId, uri);
    }

    function batchMint(address[] calldata recipients, string[] calldata uris, bool _soulbound)
        external onlyRole(MINTER_ROLE) whenNotPaused 
    {
        require(recipients.length == uris.length, "MFPNft: Length mismatch");
        require(_nextTokenId + recipients.length <= MAX_SUPPLY, "MFPNft: Exceeds max supply");
        for (uint i = 0; i < recipients.length; i++) {
            uint256 tokenId = _nextTokenId++;
            _safeMint(recipients[i], tokenId);
            _setTokenURI(tokenId, uris[i]);
            soulbound[tokenId] = _soulbound;
        }
    }

    function _update(address to, uint256 tokenId, address auth)
        internal override(ERC721, ERC721Enumerable) returns (address) 
    {
        // Block transfers for soulbound tokens (except minting/burning)
        address from = _ownerOf(tokenId);
        if (soulbound[tokenId] && from != address(0) && to != address(0)) {
            revert("MFPNft: Soulbound token is non-transferable");
        }
        return super._update(to, tokenId, auth);
    }

    function tokenURI(uint256 tokenId) public view override(ERC721, ERC721URIStorage) returns (string memory) {
        return super.tokenURI(tokenId);
    }

    function supportsInterface(bytes4 interfaceId)
        public view override(ERC721, ERC721Enumerable, ERC721URIStorage, AccessControl) returns (bool) {
        return super.supportsInterface(interfaceId);
    }

    function pause() external onlyRole(DEFAULT_ADMIN_ROLE) { _pause(); }
    function unpause() external onlyRole(DEFAULT_ADMIN_ROLE) { _unpause(); }
}
